# Time

相对时间组件

# Usage

<template>
  <Time :value="time" />
</template>
<script lang="ts">
  import { defineComponent, reactive, toRefs } from 'vue';
  import { Time } from '/@/components/Time';

  export default defineComponent({
    components: { Time },
    setup() {
      const now = new Date().getTime();
      const state = reactive({
        time: now - 60 * 3 * 1000,
      });
      return {
        ...toRefs(state),
        now,
      };
    },
  });
</script>

# Props
